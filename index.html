        // 2. ИНТЕЛЛЕКТУАЛЬНЫЙ ПАРСЕР ТЕКСТА
        function parseTestData() {
            // Убираем разрывы строк, которые разбивают слова и элементы
            const cleanText = rawDocumentText.replace(/\n|\r/g, "");
            
            // Превращаем маркеры в разделители
            const tokenized = cleanText.replace(/\/g, "||SRC:$1||");
            const parts = tokenized.split("||");
            
            let fullStringWithSources = "";
            let currentSource = "1";

            for(let part of parts) {
                if(part.startsWith("SRC:")) {
                    currentSource = part.substring(4);
                } else {
                    // Ищем буквы вариантов (a) ) и цифры вопросов (1. )
                    let chunk = part.replace(/([a-e]\) )/g, "~~$1");
                    chunk = chunk.replace(/(\d{1,3}\. )/g, "~~$1");

                    let subParts = chunk.split("~~");
                    for(let sp of subParts) {
                        let cleanSp = sp.replace(/\x0B/g, "").trim();
                        if(cleanSp) {
                            // Автоматически добавляем цитирование к каждой фразе согласно правилам!
                            fullStringWithSources += "~~" + cleanSp + "";
                        }
                    }
                }
            }

            const allElements = fullStringWithSources.split("~~").filter(e => e.trim());
            let currentQ = null;

            for(let el of allElements) {
                if (/^\d{1,3}\./.test(el)) {
                    if (currentQ) questions.push(currentQ);
                    currentQ = { question: el, options: [] };
                } else if (/^[a-e]\)/.test(el)) {
                    if (currentQ) currentQ.options.push(el);
                }
            }
            if (currentQ) questions.push(currentQ);

            // Привязываем правильные ответы
            questions.forEach((q, index) => {
                q.correct = answerKeys[index] !== undefined ? answerKeys[index] : 0;
            });
        }

        function handleAnswer(selectedIndex, selectedBtn) {
            const isCorrect = selectedIndex === questions[currentQuestionIndex].correct;
            const allBtns = DOM.optionsContainer.querySelectorAll('.option-btn');

            allBtns.forEach((btn, idx) => {
